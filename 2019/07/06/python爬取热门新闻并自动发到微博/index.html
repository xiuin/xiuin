python+微博的新闻自动发布

xiuin · 2019-7-6 · 次阅读

目标

  • 新闻源，澎湃新闻：https://www.thepaper.cn/
  • 每晚12点执行爬虫，获取到下一天要发布的内容
  • 按照评论数排序，取前12条
    • 少于15字的剔除
    • 带有“|”符号的剔除
    • 视频新闻剔除
  • 脚本一键生成发布格式
  • 如果新闻中有敏感词，需要转成图片发布

爬取热度最高的前20条新闻

# coding:utf-8

import requests
import lxml
from bs4 import BeautifulSoup
import time

# 为了解决翻页问题，我并没有直接用这个url
url = 'http://www.thepaper.cn/'
parseRetList = []

# 循环获取一天的的数据
def getAllPage(pageNum):
    for i in range(2, pageNum):
        t = time.time()
        # 精选新闻
        url = 'https://www.thepaper.cn/load_chosen.jsp?nodeids=25949&topCids=3453401,3451038,3451335,3451175,3453320,&pageidx=%d&lastTime=%d' % (i, int(round(t * 1000)))
        # 时事新闻
        # url = 'https://www.thepaper.cn/load_index.jsp?nodeids=25462,25488,25489,25490,25423,25426,25424,25463,25491,25428,68750,27604,25464,25425,25429,25481,25430,25678,25427,25422,25487,25634,25635,25600,&topCids=,3458196,3460097,3458214,3458151,3458216&pageidx=%d&lastTime=%d' % (i, int(round(t * 1000)))
        print('--------------- %d --------------- ' % i)
        respons = requests.get(url)
        respons.encoding = ('utf8')
        res = respons.text
        soup = BeautifulSoup(res, 'lxml')
        news = soup.find_all(attrs = {'class': 'news_li'})
        if len(news) == 0:
            return
        parseOnePage(news)

# 解析一页的数据
def parseOnePage(news):
    for i in range(len(news)):
        title = ''
        hot = 0
        if news[i].find(attrs = {'class': 'p_time'}):
            p_time = news[i].find(attrs = {'class': 'p_time'}).get_text()
            if p_time != '':
                continue
        if news[i].find('h2') and news[i].find('h2').find('a'):
            title = news[i].find('h2').find('a').get_text()
            # 长度太短的不要
            if len(title) < 16:
                continue
            # 有这个符号的新闻标题一般无法直接看懂
            if "|" in title:
                continue
            # 不要疑问句的新闻
            if title[-1] == '?':
                continue
        trbszan = news[i].find(attrs = {'class': 'trbszan'})
        if trbszan:
            hot = trbszan.get_text()
            if "k" in hot:
                hot = hot[:-1]
                hot = float(hot)*1000
            # 偏向于国家大事
            # if "外交部" in title:
            #     hot = int(hot) + 3000
            # elif "国" in title:
            #     hot = int(hot) + 2500
        else:
            continue
        print(title)
        print(hot)
        parseRetList.append({'title': title, 'hot': int(hot)})

# 打印排名前20的新闻
getAllPage(40)
parseRetList.sort(key=lambda k: k['hot'], reverse = True)
parseRetList = parseRetList[0:20]
for i in range(0, len(parseRetList)):
  print("[%d] %s" % (parseRetList[i]['hot'], parseRetList[i]['title']))

# 按发布格式打印排名前12的新闻
timeNow = str(time.strftime("[%Y.%m.%d]"))
info = ''
for i in range(1, 13):
  info = info + ("%d. %s\n" % (i, parseRetList[i-1]['title']))
print(info)
